> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax.

**Global Constraints + Shared Contract:** see [sp2-00-overview.md](2026-07-01-sp2-00-overview.md) (authoritative). Design: [SP-2 spec §8](../architecture/2026-07-01-sp2-intake-clarification-design.md) (Gate #1), §6.7 (MCV), §11 (lifecycle), §4.6 (folded-status idiom).

---

This phase builds **Human Gate #1 — the audited intent lock** (spec §8). It is a **fold + inline-guard**
gate over SP-2's own **`feature_contract`** aggregate (spec §4.6, §11) — **never** the built-but-unused
`state_machine/engine.py`, **never** `run_workflow_state`, **never** a document-`Stage` transition. Every
handler **folds the current status first** (`fold_feature_contract_state(stream).status`), validates the
advance **inline before appending** (mirroring `overlay/confirmation_commands.py`, incl. a no-regression
guard), and only then writes.

It ships three handlers into `intake/commands.py`:

- **`open_gate1_task`** — opens a **SEPARATE, dedicated** `CLARIFICATION`-gate confirmation task
  (`allowed_responses=[confirm,edit,reject]`), **only after MCV passes** (folded status
  `MINIMUM_CONTRACT_VALIDATED`), **`delegation_allowed=False`**, `eligible_assignees={owner}`,
  `required_inputs=[final Draft doc ref]`; and **cancels** any still-pending per-field clarification tasks
  (`cancel_tasks_on_run_advance`). Spec §8.6.
- **`confirm_contract`** — the Gate #1 `confirm`: the **`confirmer_is_requester_human`** guard
  (owner ∧ `actor_kind=="human"`; a mismatch is **denied + security-audited**, never counted); the **§8.4
  prohibited-intent screen** (authoritative block; version-drift re-check; **never fakes compliance
  approval**); the **`requires_independent_validation`** risk-flag (flag only, **no** second signer);
  hypothesis-mode candidate promotion via the document **`PRIMARY_SELECTED`** primitive; persists the full
  **confirmation record** (selected/rejected candidate doc-IDs + human edits + assumptions + ambiguity notes
  + confirmer identity); emits the frozen **`CONFIRMED_CONTRACT`** document + **`CONTRACT_CONFIRMED`** event;
  folds status → **`CONFIRMED`**. **task-version OCC** on every confirm. Spec §8.1–§8.6.
- **`request_edit`** — the Gate #1 `edit`: supersede the Draft with a **REVISED** version, capture the human
  edit, **re-run MCV**, and re-open a fresh gate task (or drop into the Refinement Loop if the edit re-opens a
  field). **task-version OCC**. Spec §8.6.

The Gate #1 `reject` response is a **requester abandonment** → SP-0 `withdraw` (existing
`("withdraw","","data_scientist","human",None)`, → `RUN_WITHDRAWN`) — **not** built here (SP-2 reuses SP-0
`withdraw`); the platform's deterministic `reject_intent` block is P8's terminal wiring.

def seed_validated_contract(db, *, run_id, request_id, draft_body, candidate_docs=0):
    """Seed a run whose contract folds to MINIMUM_CONTRACT_VALIDATED (ready for Gate #1).
    Returns (final_draft_doc_id, [candidate_doc_ids])."""
    draft_doc_id = seed_needs_clarification(
        db, run_id=run_id, request_id=request_id, draft_body=draft_body
    )
    cand_ids = [
        _freeze_draft_doc(
            db, run_id=run_id, request_id=request_id, body=draft_body, branch_role="candidate"
        )
        for _ in range(candidate_docs)
    ]
    append_fc_event(
        db, run_id=run_id, type="MINIMUM_CONTRACT_VALIDATED",
        payload={"run_id": run_id}, actor=INTAKE_SVC,
    )
    return draft_doc_id, cand_ids

    body: dict | None = None
    for e in stream:
        if e.type in ("DRAFT_CONTRACT_PRODUCED", "CONTRACT_REFINED"):
            doc_id = e.payload.get("document_id")
            body = e.payload.get("draft_body")
    return doc_id, body


def _confirmer_is_requester_human(stream, actor) -> bool:
    """§8.2 guard: confirmer_is_requester_human = actor_is_request_owner ∧ actor_kind=='human'.
    SP-0 authz only admits the data_scientist ROLE (never the specific subject), so SP-2 pins it."""
    return actor.actor_kind == "human" and actor.subject == _request_owner(stream)


def _deny_audited(conn: DbConn, cmd: Command, aggregate_id: str, reason: str) -> CommandResult:
    """A confirmer/authority denial happens INSIDE the handler (SP-0's coarse authorizer only audits
    role/kind/scope), so route it to the tamper-evident security-audit stream (§8.2) — a spoofed
    confirmer never leaves zero audit trace. Benign wrong-state / stale-OCC / prohibited-block
    denials stay unaudited (plain CommandResult)."""
    record_denial(conn, replace(cmd, aggregate_id=aggregate_id), reason)
    return CommandResult(accepted=False, aggregate_id=aggregate_id, denied_reason=reason)


def _freeze_contract_doc(
    conn: DbConn,
    *,
    run_id: str,
    request_id: str | None,
    stage: str,
    body: dict,
    branch_role: str,
    derived_from: tuple[str, ...],
    supersedes: tuple[str, ...],
    actor,
) -> str:
    """Freeze one write-once, content-hashed SP-2 contract document on the SP-0 DAG (§4.6). The body
    itself rides the emitting FC event payload (replayable, non-PII — mirrors SP-1's confirmed value);
    the document row is the DAG artifact (content-hash + lineage). governance-retained (§8.3)."""
    doc_id = mint_id("doc")
    body_bytes = json.dumps(body, sort_keys=True, separators=(",", ":")).encode("utf-8")
    append_document(
        conn,
        NewDocument(
            doc_id=doc_id,
            stage=stage,
            schema_version=1,
            branch_role=branch_role,
            content_hash=compute_content_hash(body_bytes),
            body_classification="governance-retained",
            provenance=provenance_for(stage),
            body_ref=None,
            derived_from=tuple(derived_from),
            supersedes=tuple(supersedes),
        ),
        run_id=run_id,
        request_id=request_id,
        actor=actor,
    )
    return doc_id

def _open_gate1_task(conn: DbConn, run_id: str, *, actor) -> CommandResult:
    """Open the SEPARATE, dedicated Gate-#1 confirmation task — ONLY after MCV passes (§8.6). Reused
    by both the open_gate1_task command and request_edit's gate re-open (Task 7.6)."""
    stream = load_stream(conn, "feature_contract", run_id)
    if not stream:
        return CommandResult(accepted=False, aggregate_id=run_id or "", denied_reason="unknown feature_contract")
    state = fold_feature_contract_state(stream)
    if state.status is not FeatureContractStatus.MINIMUM_CONTRACT_VALIDATED:
        return CommandResult(
            accepted=False,
            aggregate_id=run_id,
            denied_reason=f"Gate #1 requires MINIMUM_CONTRACT_VALIDATED (status={state.status.value})",
        )
    if state.open_fields:  # open_fields_empty (§6.7 #3) — never open Gate #1 on an under-specified contract
        return CommandResult(
            accepted=False,
            aggregate_id=run_id,
            denied_reason=f"open_fields not empty: {tuple(state.open_fields)}",
        )
    owner = _request_owner(stream)
    if owner is None:
        return CommandResult(accepted=False, aggregate_id=run_id, denied_reason="no request owner recorded")
    draft_doc_id, _ = _final_draft(stream)
    # Defensive close: cancel any still-pending per-field clarification tasks so none can be
    # answered behind the open gate (§8.6). After a passing MCV there should be none.
    cancel_tasks_on_run_advance(conn, run_id, reason="Gate #1 opened — pending clarification tasks cancelled")
    open_task(
        conn,
        GateTaskSpec(
            gate="CLARIFICATION",  # rides the existing CLARIFICATION gate — no new gate value (§8.6)
            required_inputs=(draft_doc_id,) if draft_doc_id else (),
            eligible_assignees={"role": "data_scientist", "subject": owner},
            allowed_responses=("confirm", "edit", "reject"),
            run_id=run_id,
            delegation_allowed=False,  # the author-owned intent lock (§8.2)
        ),
        actor,
    )
    return CommandResult(accepted=True, aggregate_id=run_id)


def open_gate1_task(conn: DbConn, cmd: Command) -> CommandResult:
    run_id = cmd.args.get("run_id") or cmd.aggregate_id

        conn, run_id=run_id, type="CONTRACT_REFINED",
        payload={
            "document_id": revised_doc_id,
            "draft_body": revised,
            "human_edits": [human_edit],
            "reopened": reopened,
        },
        actor=cmd.actor,
        expected_version=stream[-1].stream_version,
    )
    if reopened:
        # The edit re-introduced an open field → back into the Refinement Loop (§6.6); no gate re-open.
        return CommandResult(accepted=True, aggregate_id=run_id)
    # Re-run MCV on the revised Draft; if it still passes, re-open a fresh Gate #1 task (§8.6).
    mcv = minimum_contract_validated(conn, run_id, actor=cmd.actor)
    if mcv.accepted:
        _open_gate1_task(conn, run_id, actor=cmd.actor)
    return CommandResult(accepted=True, aggregate_id=run_id)
